# claculator
easy claculator

const claculator = {
    add: function (a, b) {
        console.log(a + b)
        
    },
    minus: function(a, b) {
        console.log(a - b)

    },
    multiplied: function(a, b) {
        console.log(a * b)

    },
    divided: function(a, b) {
        console.log(a / b)

    },
    squared: function(a, b) {
        console.log(a ** b)

    }
    
}

claculator.add(5, 4)
claculator.minus(5, 4)
claculator.multiplied(5, 4)
claculator.divided(5, 4)
claculator.squared(5, 4)
